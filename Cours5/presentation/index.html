Schema

Définit la structure d'une collection de MongoDB.

var mongoose = require('mongoose');

var catSchema = new mongoose.Schema({
  name: String,
  furColor: String,
  birthDate: Date,
  owner: String
});

Model

Définition d'un document basé sur un schéma.

var Cat = mongoose.model('Cat', catSchema);
