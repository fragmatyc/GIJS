Importation et connexion

var mongoose = require('mongoose');
mongoose.connect('mongodb://localhost/test');

Schema

Définit la structure d'une collection de MongoDB.

var mongoose = require('mongoose');

var catSchema = new mongoose.Schema({
  name: String,
  furColor: String,
  birthDate: Date,
  owner: String
});

Model

Définition d'un document basé sur un schéma.

var Cat = mongoose.model('Cat', catSchema);

Instance

Création d'un object (document) à persister.

var mimi = new Cat({
  name: "Mimi",
  furColor: "White",
  birthDate: new Date(),
  owner: "Sylvain"
});

Persistence

mimi.save(function (err) {
  if (err) {
    console.log("Une erreur nous empêche de sauvegarder le chat.")
  } else {
    console.log("Chat sauvegardé!");
  }
});

Requête

Cat.find({}, function (err, dataFound) {
  console.log(dataFound);
});
